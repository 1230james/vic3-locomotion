LLWA_company_turnpike = {
    icon       = "gfx/interface/icons/company_icons/LLWA_overland_logo.dds"
	background = "gfx/interface/icons/company_icons/company_backgrounds/LLWA_comp_illu_roads.dds"
    
    category = shopkeeper_owned
    flavored_company = no
    uses_dynamic_naming = yes
    
    dynamic_company_type_names = {
        dynamic_company_type_company
        dynamic_company_type_guild
    }
    
    building_types = {
        LLWA_building_roadway
        LLWA_building_waterway
        LLWA_building_riverway
    }

    extension_building_types = {
        building_railway
    }

    possible_prestige_goods = {
        # LLWA_pg_express_service # DO NOT USE - prestige goods do NOT work for local goods
    }

    prestige_goods_trigger = {} # default is yes
    
    potential = {
        has_technology_researched = urbanization
    }
    
    attainable = {
        always = yes
    }
    
    possible = {
        any_scope_state = {
            any_scope_building = {
                OR = {
                    is_building_type = LLWA_building_roadway
                    is_building_type = LLWA_building_waterway
                    is_building_type = LLWA_building_riverway
                }
                level >= 5
            }
        }
    }

    prosperity_modifier = {
        state_infrastructure_from_population_mult     = 0.05
        state_infrastructure_from_population_max_mult = 0.05
    }

    ai_will_do = {
        num_companies <= {
            add      = max_num_companies
            subtract = 2
        }
        any_scope_state = {
            any_scope_building = {
                OR = {
                    is_building_type = LLWA_building_waterway
                    is_building_type = LLWA_building_riverway
                }
                level >= 3
            }
        }
    }
    
    ai_construction_targets = {
        LLWA_building_riverway = {
			level = 5
			state_trigger = {
				is_incorporated = yes
                infrastructure_delta < 10
			}
		}
        LLWA_building_waterway = {
			level = 5
			state_trigger = {
				is_incorporated = yes
                infrastructure_delta < 10
                owner = {
                    NOT = {
                        any_scope_state = {
                            LLWA_st_on_river = yes
                        }
                    }
                }
			}
		}
    }
    
    ai_weight = {
        value = 0.5
    }
}
