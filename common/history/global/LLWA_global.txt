        limit = {
            OR = {
                has_building = building_railway
                has_building = LLWA_building_waterway
                has_building = LLWA_building_roadway
            }
        }
        
        # Create the building
        if = {
            limit = {
                calc_true_if = {
                    amount = 1
                    has_building = building_railway
                    has_building = LLWA_building_waterway
                    has_building = LLWA_building_roadway
                }
            }
            create_building = {
                building   = LLWA_building_freight_depot
                subsidized = no
                reserves   = 1
                level      = 2
            }
        }
        else_if = {
            limit = {
                calc_true_if = {
                    amount = 2
                    has_building = building_railway
                    has_building = LLWA_building_waterway
                    has_building = LLWA_building_roadway
                }
            }
            create_building = {
                building   = LLWA_building_freight_depot
                subsidized = no
                reserves   = 1
                level      = 4
            }
        }
        else = {
            create_building = {
                building   = LLWA_building_freight_depot
                subsidized = no
                reserves   = 1
                level      = 6
            }
        }
        
        # Update PMs if necessary
        if = {
            limit = {
                owner ?= {
                    has_technology_researched = international_trade
                }
            }
            activate_production_method = {
                building_type     = LLWA_building_freight_depot

            }
